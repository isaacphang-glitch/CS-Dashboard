function doGet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var rosterSheet = ss.getSheetByName('Roster'); var rosterData = rosterSheet.getDataRange().getValues(); var roster = []; var award = ''; for (var i = 1; i < rosterData.length; i++) { var name = String(rosterData[i][0]).trim(); if (!name) continue; if (name.toLowerCase() === '__award__') { award = String(rosterData[i][1]).trim(); } else { roster.push({ name: name, status: String(rosterData[i][1]).trim().toLowerCase(), role: String(rosterData[i][2]).trim().toUpperCase() }); } } var logSheet = ss.getSheetByName('Log'); var logData = logSheet.getDataRange().getValues(); var now = new Date(); var todayStr = Utilities.formatDate(now, Session.getScriptTimeZone(), 'yyyy-MM-dd'); var weekStart = new Date(now); weekStart.setDate(now.getDate() - ((now.getDay() + 6) % 7)); var weekStartStr = Utilities.formatDate(weekStart, Session.getScriptTimeZone(), 'yyyy-MM-dd'); var monthStartStr = Utilities.formatDate(new Date(now.getFullYear(), now.getMonth(), 1), Session.getScriptTimeZone(), 'yyyy-MM-dd'); var daily = {}, weekly = {}, monthly = {}; for (var j = 1; j < logData.length; j++) { var raw = logData[j][0]; if (!raw) continue; var agent = String(logData[j][1]).trim(); var bookings = parseInt(logData[j][2]) || 0; if (!agent) continue; var entryStr; if (raw instanceof Date) { entryStr = Utilities.formatDate(raw, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } else { var parts = String(raw).split('/'); if (parts.length !== 3) continue; var d = new Date(parseInt(parts[2]), parseInt(parts[1]) - 1, parseInt(parts[0])); entryStr = Utilities.formatDate(d, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } if (!daily[agent]) daily[agent] = 0; if (!weekly[agent]) weekly[agent] = 0; if (!monthly[agent]) monthly[agent] = 0; if (entryStr === todayStr) daily[agent] += bookings; if (entryStr >= weekStartStr) weekly[agent] += bookings; if (entryStr >= monthStartStr) monthly[agent] += bookings; } var members = roster.map(function(m) { return { name: m.name, status: m.status, role: m.role, daily: daily[m.name] || 0, weekly: weekly[m.name] || 0, monthly: monthly[m.name] || 0 }; }); return ContentService .createTextOutput(JSON.stringify({ members: members, award: award })) .setMimeType(ContentService.MimeType.JSON); }
